001package uniandes.unacloud.web.utils.java;
002
003import java.io.UnsupportedEncodingException;
004import java.security.MessageDigest;
005import java.security.NoSuchAlgorithmException;
006
007import org.apache.commons.lang3.RandomStringUtils;
008
009import com.sun.org.apache.xerces.internal.impl.dv.util.Base64;
010
011/**
012 * Class used to create random hash, numbers and strings
013 * @author CesarF
014 *
015 */
016public class Hasher {
017        private static final String charset = "abcdefghijklmnopqrstuvwxyz0123456789";
018        private static final String HASH_ALGORITHM = "SHA-256";
019        
020        /**
021         * Method to create a hash based in a text using sha256
022         * @param text
023         * @return hash String
024         * @throws NoSuchAlgorithmException
025         * @throws UnsupportedEncodingException
026         */
027        public static String hashSha256(String text) throws NoSuchAlgorithmException, UnsupportedEncodingException{
028                MessageDigest md = MessageDigest.getInstance(HASH_ALGORITHM);
029                md.update(text.getBytes("UTF-8"));
030                return Base64.encode(md.digest());              
031        }
032        
033        /**
034         * Using to create a randomString based in a length sent by user
035         * @param ln
036         * @return random string based in charset
037         */
038        public static String randomString(int ln){
039                return RandomStringUtils.random(ln, charset.toCharArray());
040        }
041}
